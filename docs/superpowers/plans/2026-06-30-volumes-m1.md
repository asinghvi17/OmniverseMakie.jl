# IndeX enablement — process-global, set ONCE, BEFORE the first ovrtx_create_renderer.
# Volume rendering routes UsdVol prims to NVIDIA IndeX Direct, whose loader resolves the
# carb token ${omni.index.libs} (Kit-only, absent in standalone ovrtx) to find its libs.
# We register that token by feeding ovrtx a carb config via CARB_FRAMEWORK_CONFIG_NAME.
const _INDEX_ENABLED = Ref(false)

const _ensure_index = Base.OncePerProcess{Bool}() do
    cfg = get(ENV, "OMNIVERSEMAKIE_OVRTX_CONFIG", "")
    libs = get(ENV, "OMNIVERSEMAKIE_INDEX_LIBS", "")
    (isempty(cfg) && isempty(libs)) && return false           # disabled → zero overhead
    try
        ovrtx_bin = dirname(ENV["OVRTX_LIBRARY_PATH"])         # <ovrtx>/bin (CARB_APP_PATH)
        config_file = isempty(cfg) ? _synth_config(ovrtx_bin, libs) : cfg
        # carb force-joins CARB_FRAMEWORK_CONFIG_NAME onto CARB_APP_PATH and appends .config.json
        ENV["CARB_FRAMEWORK_CONFIG_NAME"] = _relpath_for_carb(ovrtx_bin, config_file)  # Step-2 form
        _INDEX_ENABLED[] = true
        return true
    catch e
        @warn "OmniverseMakie: could not enable IndeX volume rendering; volumes will be unavailable" exception=e maxlog=1
        return false
    end
end

_index_enabled() = _INDEX_ENABLED[]

function _colormap_rgba_points(colormap, colorrange)
    cs = Makie.to_colormap(colormap)
    n  = length(cs)
    io = IOBuffer()
    for (i, c) in enumerate(cs)
        t = (i - 1) / (n - 1)
        # rgbaPoints: position then RGBA (opacity ramps with t so low density is transparent)
        print(io, "(", t, ", ", Float32(Makie.red(c)), ", ", Float32(Makie.green(c)),
              ", ", Float32(Makie.blue(c)), ", ", Float32(t), ")")
        i < n && print(io, ", ")
    end
    return String(take!(io))
end
